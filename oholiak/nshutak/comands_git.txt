cd </шлях> -- відкриває певну директорію, яку вказати
ls -- показує всі файли у вибраній директорії
pwd -- показує, де ми зараз знаходимось (в якій папці)


            КОМАНДИ НА ЛОКАЛЬНОМУ КОМП'ЮТЕРІ (РОБОТА З ЛОКАЛЬНИМ РЕПОЗИТОРІЄМ)
git --help
git clear -- очищує консоль
git init -- ініціалізує нове сховище коду (тобто створює, перетворює просту папку в репозиторій)

git status -- перевіряє, які зміни відбулися
git add <ім'я файлу> -- додає файл у репозиторій
git add . -- додає всі файли чи зміни, що відбулися
git commit -m <текст повідомлення> -- додає коміт з повідомленням, де ми детально описуємо, які зміни ми зробили
git commit -am <текст повідомлення> -- команда 3-в-1, додає всі зміни, робить коміт, додає меседж до коміту

git diff -- показує різницю
git log -- показує історію змін на вітці, історію комітів, до кожного з яких можна повернутися в будь-який момент
git log --graph 

git show <назва коміту> -- показує, переходить певний коміт


         КОМАНДИ НА ВІДДАЛЕНОМУ КОМП'ЮТЕРІ (РОБОТА З ВІДДАЛЕНИМ РЕПОЗИТОРІЄМ github)
git clone <посилання> -- клонує віддалений репозиторій собі на локальний комп'ютер
git checkout -b <назва вітки> -- git створює нову вітку 
git checkout <назва вітки> -- переходить в нову вітку
git branch -D <назва вітки> -- видаляє вітку

git merge <назва вітки> -- додає, з'єднює нову вітку з головною віткою main
git push origin <назва вітки> -- відправляє у віддалений репозиторій на Github вітку та зміни у ній
git pull origin <назва вітки> -- "забирає" зміни з віддаленого репозиторія на локальну вітку.
